// Load courses from markdown files async function loadCourses() { try { const response = await fetch('/content/courses.json'); if (!response.ok) throw new Error('Courses not found'); const courses = await response.json(); return courses; } catch (error) { console.log('Loading individual course files...'); // Fallback: Try to load each .md file const courseFiles = [ 'professional-hairdressing' ]; const courses = []; for (const file of courseFiles) { try { const mdResponse = await fetch(`/content/courses/${file}.md`); if (mdResponse.ok) { const mdText = await mdResponse.text(); // Parse frontmatter (simplified) const titleMatch = mdText.match(/title: "(.+?)"/); const descMatch = mdText.match(/description: "(.+?)"/); const durationMatch = mdText.match(/duration: "(.+?)"/); const youtubeMatch = mdText.match(/youtube: "(.+?)"/); if (titleMatch) { courses.push({ id: courses.length + 1, title: titleMatch[1], description: descMatch ? descMatch[1] : '', duration: durationMatch ? durationMatch[1] : '', youtubeUrl: youtubeMatch ? youtubeMatch[1] : '', thumbnail: "https://placehold.co/600x338/fdf2f8/9b2d6e?text=Course+Video" }); } } } catch (err) { console.log(`Could not load ${file}`); } } return courses; } } // Render courses async function renderCourses() { const container = document.getElementById('coursesContainer'); if (!container) return; const courses = await loadCourses(); if (courses.length === 0) { container.innerHTML = '

No courses found. Add courses in the CMS!

'; return; } container.innerHTML = courses.map(course => `

${course.title}

${course.duration}

${course.description}

`).join(''); } // Call render function when page loads renderCourses();
